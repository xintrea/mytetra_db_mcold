class MyPanel(wx.Panel):

        self.frame = parent

        #self.Text = wx.StaticText(self, 1, pos=(5, 5), size = (400, 150))
        self.Btn = wx.Button(self, 5, pos=(5, 5))

        self.TT = wx.TextCtrl(self, 3, pos = (5, 150), size = (200, 20))
