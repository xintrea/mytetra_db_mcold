<SQL>

<getid_qjm>

[Out cost]

begin

begin

Select q.id Into :idq

From qjm_product q

, pdm_shipproduct p

Where p.id = :idp

And p.scode = q.scode;

end;

</getid_qjm>


<SetFlt>

[out Filter$Flt_idProduct, Filter$Flt_idProduct_HL]

begin

:Filter$Flt_idProduct := :idqProduct;

:Filter$Flt_idProduct_HL := QJM_ProductAPI.GetHeadLine(:idqProduct);

end;

</SetFlt>

</SQL>
