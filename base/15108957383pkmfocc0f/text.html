// Задача:

// заведение массива элементов с выводом в обратном порядке

// только положительных элементов


#include "stdafx.h"

#include <iostream>

#include <vector>


using namespace std;


void main()

{

int n;

char x;

cin >> n;

vector <int> a;


for (int i = 0; i < n; i++) {

int temp;

cin >> temp;

if (temp > 0) {

a.push_back(temp);

}


}


// вывод в обратном порядке положительных элементов

for (int j = a.size() - 1; j >= 0; j--) {

cout << a[j] << endl;

}

cout << "-----------";


cin >> x;

//return 0;




}
