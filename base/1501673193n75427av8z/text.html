#!/bin/bash



redEcho(){

echo -ne "\033[1;32m" # устанавливаем формат вывода - зеленый цвет - цвет задается последней цифрой

echo "$@" # выводим список всех параметров функции в столбец

echo " "

echo -n "$*" # выводим список параметров функции в строку

echo -e "\033[0m" # возвращаем стандартный формат вывода

echo " "

echo "I'm not coloured string" # эта строка со стандартным выводом

}


redEcho $0 $1 $2

~
