def hide(self, widget, data=None):
self.button_toHide.hide()

self.box = gtk.HBox(False, 0) # expand = False

self.box.pack_start(self.button, False, True, 0) # homogeneous=False
self.box.pack_start(self.button_toHide, False, True, 0)
self.box.pack_start(self.button_Hide, False, True, 0)

self.window.add(self.box)

self.box.show()
# Последним щагом будет отображение созданной кнопки.
self.button.show()

self.button_toHide.show()

self.button_Hide.show()

# и окна
self.window.show()
