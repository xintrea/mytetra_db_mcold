Подготовьте отчет для отдела HR, в котором показываются номер отдела, фамилия служащего и фамилии всех служащих, работающих в одном отделе с данным служащим. Дайте столбцам соответствующие имена. Сохраните скрипт в файле lab_06_06.sql.


select d.department_name department

, e.last_name employee

, e.department_id

, c.last_name colleague

, c.department_id

from employees e

, departments d

, employees c

where e.department_id = d.department_id

and c.department_id = d.department_id

and not e.employee_id = c.employee_id -- убираем сочетание самого сотрудника с самим собой

order by e.last_name asc, c.last_name asc
