"""
Строки: объединение и разъединение
"""


# 1. splitlines
# аргумент keepends - позволяет включать символ окончания строки \n
ss = "hunny\npussy\npitty"

print(ss.splitlines(keepends=True))

# -------------
# ['hunny\n', 'pussy\n', 'pitty']
# -------------


# 2. rsplit

print(ss.rsplit(sep='\n', maxsplit=1))
# -------------
# ['hunny\npussy', 'pitty']
# -------------
