SendMode Input

KeysCount := 0


global Array := Object()




;==========================================================

^!sc02D:: ; Ctrl + Alt + X

MsgBox Скрипт "Macro" будет завершён.

ExitApp

Return





^F11:: ; Ctrl + F11 начало записи макроса

Rec := 1 ; флаг записи

KeysCount := 0 ; количество клавиш (команд) в макросе



Return


^LButton::

;MsgBox "Количество кнопок - " %KeysCount%

if(KeysCount > 0)

; собираем в 1 строку

Loop %KeysCount%

{

element := Macro%A_Index%

text .= element

}

; записываем в массив

Array.Insert(text)

; обнуляем количество элементов

KeysCount := 0

return




^RButton::

for index, element in Array

{

Send %element%

}

return


Add(Key) ; запоминание нажатой клавиши

{

global

If Rec = 1

{

KeysCount += 1

Macro%KeysCount% := Key


}

}
