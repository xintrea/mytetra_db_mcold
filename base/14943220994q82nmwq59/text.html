create table lobs_tab (

id number(5),

clob_col clob,

nclob_col nclob,

blob_col blob,

bfile_col bfile

);


select table_name, column_name, segment_name

from user_lobs

where table_name = 'LOBS_TAB';


insert into lobs_tab (id, clob_col, nclob_col, blob_col, bfile_col)

values (1, null, empty_clob(), empty_blob(), null);


update lobs_tab set clob_col = 'привет, мир' where id = 1;


select length(clob_col),

dbms_lob.getlength(clob_col),

'"' || clob_col || '"' from lobs_tab

where clob_col like '%мир%';
