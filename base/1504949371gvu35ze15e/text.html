"""
Управляющие последовательности
"""


### Возврат каретки
# \r
# Возвращается на начало текущей строки

ss = "hunny\r"

print(ss+'testy')
# ----------
# testy             т.е. не происходит записи 'hunny'
# ----------


### Backspace
# \b
# Возвращается на начало текущей строки

ss = "hunny\b"
print(ss)

# ----------
#   hunn           т.е. удаляет последний символ
# ----------
