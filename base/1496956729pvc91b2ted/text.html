вывести список подразделений, количество сотрудников в которых больше 5, в виде сочетаний полей:

- имя подразделения

- количество сотрудников

+ результирующая строка "ИТОГО:" общее кол-во



with emp_dept as (select e.department_id di

, count(e.department_id) cdi

from employees e

group by e.department_id

having count(e.department_id) > 5

)


select d.department_name

, ed.cdi

from emp_dept ed

, departments d

where ed.di = d.department_id

union

select 'ИТОГО:'

, count(e.department_id)

from employees e
