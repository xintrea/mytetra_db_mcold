create or replace procedure AddMarks(p_admno number, p_marks number)

is

v_marks marks_array;

begin

select marks into v_marks

from students

where admno = p_admno;

v_marks.extend();

v_marks(v_marks.last) := p_marks;

update students set marks = v_marks

where admno = p_admno;

commit;

end;
