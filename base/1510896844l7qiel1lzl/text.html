#include "stdafx.h"

#include <iostream>

#include <vector>


using namespace std;



void main()

{

int n;

int i_min, i_max;

int t_min, t_max;

char x;

cin >> n;

vector <int> a;


for (int i = 0; i < n; i++) {

int temp;

cin >> temp;

//if (temp > 0) {

a.push_back(temp);

}


t_min = a[0];

i_min = 0;

for (int i = 0; i < a.size(); i++) {

if (a[i] < t_min) {

t_min = a[i];

i_min = i;

}

}


t_max = a[0];

i_max = 0;

for (int i = 0; i < a.size(); i++) {

if (a[i] > t_max) {

t_max = a[i];

i_max = i;

}

}


// делаем замену

a[i_min] = t_max;

a[i_max] = t_min;


for (auto j : a) {

cout << j << " ";

}

cout << endl << "---------" << endl;

cin >> x;

}
