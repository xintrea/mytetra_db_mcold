2.      Напишите запрос для вывода общего количества служащих и количества служащих, нанятых в 2001, 2002, 2003 и 2004 годах. Дайте соответствующие заголовки столбцам.


select count(distinct e.employee_id) "All"

, count(distinct e2.employee_id) "2001"

, count(distinct e3.employee_id) "2002"

, count(distinct e4.employee_id) "2003"

, count(distinct e5.employee_id) "2004"

from employees e

, employees e2

, employees e3

, employees e4

, employees e5

where extract(YEAR from e2.hire_date) = 2001

and extract(YEAR from e3.hire_date) = 2002

and extract(YEAR from e4.hire_date) = 2003

and extract(YEAR from e5.hire_date) = 2004
